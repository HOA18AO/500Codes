class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        int dp[amount+1];
        dp[0] = 0;
        for(int i=1 ; i<=amount ; i++)
        {
            dp[i] = amount+1;
            for(int c : coins)
                if(i >= c)
                    dp[i] = min(dp[i], 1 + dp[i-c]);
        }
        if(dp[amount] > amount)
            return -1;
        return dp[amount];
    }
};
