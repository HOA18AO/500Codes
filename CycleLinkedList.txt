/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool hasCycle(ListNode *head) {
        if(head == NULL)
            return false;
        ListNode *rabbit, *turtle;
        rabbit = head;
        turtle = rabbit;
        while(rabbit!=NULL and rabbit->next!=NULL)
        {
            rabbit = rabbit->next->next;
            turtle = turtle->next;
            if(rabbit == turtle)
                return true;
        }
        return false;
    }
};
