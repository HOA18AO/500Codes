class Solution {
public:
    int singleNumber(vector<int>& nums) {
        int ans = 0;
        map<int,bool> check;
        for(int i : nums)
        {
            if(check[i] == false)
            {
                ans += i;
                check[i] = true;
            }
            else
                ans -= i;
        }
        return ans;
    }
};
