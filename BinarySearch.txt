class Solution {
public:
    int search(vector<int>& nums, int target) {
        int head, tail, mid;
        head = 0;
        tail = nums.size()-1;
        while(head <= tail)
        {
            mid = (head + tail)/2;
            if(nums[mid] == target)
                return mid;
            else
                if(nums[mid] < target)
                    head = mid+1;
                else
                    tail = mid-1;
        }
        return -1;
    }
};
