class Solution {
public:
    bool isAnagram(string s, string t) {
        if(s.size()!=t.size())
            return false;
        if(s.size() == 0)
            return false;
        map<char,int> check;
        for(int i=0 ; i<s.size() ; i++)
        {
            check[s[i]]++;
            check[t[i]]--;
        }
        for(char i='a' ; i<='z' ; i++)
            if(check[i] != 0)
                return false;
        return true;
    }
};

// s and t consist of lowercase English letters
