class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int buy, sell, pro;
        pro = 0;
        buy = 99999;
        for(int i=0 ; i<prices.size() ; i++)
        {
            sell = prices[i];
            if(prices[i] < buy)
                buy = prices[i];
            pro = max(pro, sell-buy);
        }
        return pro;
    }
    
};
