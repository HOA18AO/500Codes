// The API isBadVersion is defined for you.
// bool isBadVersion(int version);

class Solution {
public:
    
    // isBadVersion(int);
    int firstBadVersion(int n) {
        int head, tail, mid;
        head = 1;
        tail = n;
        while(head <= tail)
        {
            mid = head + (tail-head)/2;
            if(isBadVersion(mid))
            {
                if(!isBadVersion(mid-1))
                    return mid;
                else
                    tail = mid;
            }
            else
                head = mid+1;
            
        }
        return tail;
    }
};
