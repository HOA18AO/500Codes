class Solution {
public:
    int missingNumber(vector<int>& nums) {
        vector<int> a(10000);
        int ans;
        for(int i : nums)
            a[i]++;
        for(int i=0 ; i<nums.size() ; i++)
            if(a[i] == 0)
            {
                ans = i;
                break;
            }
        return ans;
    }
};
